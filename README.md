```
estimateMu = function(points) {
    rows = dim(points)[1]
    cols = dim(points)[2]
    mu = matrix(NA, 1, cols)
    for (col in 1:cols) {

    }
    return(mu)
}
```

```
estimateCovarianceMatrix = function(points, mu) {
    rows = dim(points)[1]
    cols = dim(points)[2]
    covar = matrix(0, cols, cols)
    for (i in 1:rows) {
        covar = covar + (t(points[i,] - mu) %*% (points[i,] - mu)) / (rows - 1)
    }
    return(covar)
}
```

```
sg.ADALINE <- function(xl, eta = 1, lambda = 1/6)
{
  l <- dim(xl)[1]
  n <- dim(xl)[2] - 1
  w <- c(1/2, 1/2, 1/2)
  iterCount <- 0
  # –Ω–∞—á–∞–ª—å–Ω–æ–µ Q
  Q <- 0
  for (i in 1:l)
  {
    ## —Å—á–∏—Ç–∞–µ–º —Å–∫–∞–ª—è—Ä–Ω–æ–µ –ø—Ä–æ–∏–∑–≤–µ–¥–µ–Ω–∏–µ <w,x>

    ## —Å—á–∏—Ç–∞–µ–º –æ—Ç—Å—Ç—É–ø 
    margin <- wx * xl[i, n + 1]
    Q <- Q + lossQuad(margin)
  }
  repeat
  {
    
    margins <- array(dim = l)
    
    for (i in 1:l)
    {
      xi <- xl[i, 1:n]
      yi <- xl[i, n + 1]
      margins[i] <- crossprod(w, xi) * yi
    }
    # –≤—ã–±—Ä–∞—Ç—å –æ–±—ä–µ–∫—Ç—ã —Å –æ—à–∏–±–∫–æ–π
    errorIndexes <- which(margins <= 0)
    if (length(errorIndexes) > 0)
    {
      # –≤—ã–±—Ä–∞—Ç—å —Å–ª—É—á–∞–π–Ω—ã–π –∏–Ω–¥–µ–∫—Å –∏–∑  –æ—à–∏–±–æ–∫
      i <- sample(errorIndexes, 1)
      iterCount <- iterCount + 1
      xi <- xl[i, 1:n]
      yi <- xl[i, n + 1]
      wx <- sum(w * xi)
      margin <- wx * yi
      # –≤—ã—Å—á–∏—Ç—ã–≤–∞–µ–º –æ—à–∏–±–∫—É
      ex <- lossQuad(margin)
      eta <- 1 / sqrt(sum(xi * xi))
      w <- w - eta * (wx - yi) * xi
      # –†–∞—Å—á–∏—Ç—ã–≤–∞–µ–º –Ω–æ–≤–æ–µ  Q
      Qprev <- Q

    else

      break
    }
  }

}
```
